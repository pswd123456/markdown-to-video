# Example 1: Standard Layout (Title + Center Content)
# Input: "Show a title 'Binary Search' at the top, and a number array in the center."
from manim import *

class StandardLayoutScene(Scene):
    def construct(self):
        # 1. Title (Always fixed to top edge with buffer)
        title = Text("二分查找", font="Noto Sans CJK SC", font_size=48, color=BLUE)
        title.to_edge(UP, buff=0.5)
        
        # 2. Main Content (Centered by default)
        content = Text("[1, 3, 5, 7, 9]", font="Noto Sans CJK SC", font_size=60)
        content.move_to(ORIGIN) 
        
        # 3. Animation Sequence
        self.play(Write(title))
        self.play(FadeIn(content, shift=UP)) # Subtle entry animation
        self.wait(2)

# Example 2: Flow & Relations (Relative Positioning)
# Input: "Show the data flow from Client to Server with an arrow."
from manim import *

class FlowScene(Scene):
    def construct(self):
        # 1. Define Entities
        # Use VGroup to encapsulate Text + Shape if needed
        client = Text("Client", font="Noto Sans CJK SC", color=TEAL)
        server = Text("Server", font="Noto Sans CJK SC", color=ORANGE)
        
        # 2. Layout Strategy: Arrange horizontally with sufficient buffer
        # This prevents overlap regardless of text length
        group = VGroup(client, server)
        group.arrange(RIGHT, buff=4.0)
        group.move_to(ORIGIN)
        
        # 3. Dynamic Connector (Arrow relies on object boundaries)
        arrow = Arrow(start=client.get_right(), end=server.get_left(), color=WHITE)
        label = Text("Request", font="Noto Sans CJK SC", font_size=24).next_to(arrow, UP)
        
        # 4. Sequence
        self.add(client, server)
        self.play(GrowArrow(arrow), Write(label))
        self.play(Indicate(server)) # Visual feedback for emphasis
        self.wait(2)
